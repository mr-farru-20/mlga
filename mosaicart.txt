Mosaic art 
 
import cv2 
import numpy as np 
import matplotlib.pyplot as plt 
 
from google.colab import files 
uploaded = files.upload() 
image_path = next(iter(uploaded)) 
 
img = cv2.imread(image_path) 
img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB) 
img = cv2.resize(img, (400, 400)) 
 
tile_size = 10 # Size of each mosaic tile (adjust for density) 
h, w, _ = img.shape 
 
# blank canvas 
mosaic = np.zeros_like(img) 
 
# Looping to make blocks 
for y in range(0, h, tile_size): 
   for x in range(0, w, tile_size): 
 
       tile = img[y:y+tile_size, x:x+tile_size] 
 
       avg_color = tile.mean(axis=(0, 1)).astype(np.uint8) 
 
       mosaic[y:y+tile_size, x:x+tile_size] = avg_color 
 
plt.figure(figsize=(8, 8)) 
plt.imshow(mosaic) 
plt.axis('off') 
plt.title("Mosaic Art Effect") 
plt.show()
